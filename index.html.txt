<script>
/** --------- 定数・状態 --------- */
const SAFE_THRESHOLD_M = 5.0;
const CAMERA_MODELS = {
  "iPhone 16 標準広角": { focal: 4.25, sensor_h: 7.2 },
  "iPhone 16 超広角":   { focal: 2.2,  sensor_h: 4.2 },
  "iPhone 16 望遠(Pro)":{ focal: 9.0,  sensor_h: 4.8 },
};

const els = {
  file: document.getElementById('file'),
  photo: document.getElementById('photo'),
  overlay: document.getElementById('overlay'),
  stage: document.getElementById('stage'),
  mode: document.getElementById('mode'),
  model: document.getElementById('model'),
  distance: document.getElementById('distance'),
  focal: document.getElementById('focal'),
  sensorH: document.getElementById('sensorH'),
  pitch: document.getElementById('pitch'),
  guide: document.getElementById('guide'),
  result: document.getElementById('result'),
  btnCalc: document.getElementById('btnCalc'),
  btnReset: document.getElementById('btnReset'),
  btnCaptureRef: document.getElementById('btnCaptureRef'),
  btnClearCal: document.getElementById('btnClearCal'),
};

let clicks = [];      // 対象の上下2点（原画像座標）
let refClicks = [];   // 参照物の上下2点（原画像座標）
let capturingRef = false;
let calScale = 1.0, calBias = 0.0;

/** --------- 初期化 --------- */
(function init(){
  // 機種候補
  for (const name of Object.keys(CAMERA_MODELS)) {
    const opt = document.createElement('option');
    opt.value = name; opt.textContent = name;
    els.model.appendChild(opt);
  }
  els.model.value = "iPhone 16 標準広角";
  applyModel(els.model.value);

  els.model.addEventListener('change', e => applyModel(e.target.value));
  els.file.addEventListener('change', onFile);
  window.addEventListener('resize', drawOverlay);
  els.overlay.addEventListener('click', onOverlayClick);

  els.btnCalc.addEventListener('click', onCalc);
  els.btnReset.addEventListener('click', resetAll);
  els.btnCaptureRef.addEventListener('click', startCaptureRef);
  els.btnClearCal.addEventListener('click', () => { calScale = 1.0; calBias = 0.0; toast("校正をクリアしました（cal_scale=1.0, cal_bias=0.0）"); });

  updateGuide();
})();

function applyModel(modelName){
  const spec = CAMERA_MODELS[modelName];
  if (!spec) return;
  els.focal.value = spec.focal;
  els.sensorH.value = spec.sensor_h;
}

/** --------- 画像読み込み --------- */
function onFile(){
  const f = els.file.files?.[0];
  if (!f) return;
  const url = URL.createObjectURL(f);
  els.photo.onload = () => {
    fitOverlayToPhoto();
    clicks = []; refClicks = []; capturingRef = false;
    updateGuide();
    drawOverlay();
  };
  els.photo.src = url;
}

/** --------- 表示スケール計算 --------- */
function fitOverlayToPhoto(){
  // overlay を photo と同じ CSS サイズに合わせる
  const rect = els.photo.getBoundingClientRect();
  els.overlay.width = rect.width;
  els.overlay.height = rect.height;
  els.overlay.style.width = rect.width + 'px';
  els.overlay.style.height = rect.height + 'px';
  els.overlay.style.left = els.photo.offsetLeft + 'px';
  els.overlay.style.top = els.photo.offsetTop + 'px';
}

function drawOverlay(){
  if (!els.photo.src) return;
  fitOverlayToPhoto();
  const ctx = els.overlay.getContext('2d');
  ctx.clearRect(0,0,els.overlay.width,els.overlay.height);

  const drawPts = (pts, color) => {
    ctx.strokeStyle = color; ctx.fillStyle = color; ctx.lineWidth = 2;
    pts.forEach((p, i) => {
      const [dx, dy] = dispXY(p);
      ctx.beginPath();
      ctx.arc(dx, dy, 5, 0, Math.PI*2);
      ctx.stroke();
      ctx.font = 'bold 14px system-ui, -apple-system';
      ctx.fillText(i===0?'①':'②', dx+10, dy+4);
    });
    if (pts.length === 2){
      const [a,b] = pts.map(dispXY);
      ctx.setLineDash([6,4]);
      ctx.beginPath(); ctx.moveTo(a[0],a[1]); ctx.lineTo(b[0],b[1]); ctx.stroke();
      ctx.setLineDash([]);
    }
  };
  drawPts(clicks, '#ef4444');     // 赤
  drawPts(refClicks, '#fb923c');  // 橙
}

/** --------- クリック処理 --------- */
function onOverlayClick(ev){
  if (!els.photo.src) return;
  const [xo, yo] = originalXY(ev);
  if (capturingRef){
    if (refClicks.length >= 2) return;
    refClicks.push([xo, yo]);
    drawOverlay();
    if (refClicks.length === 2) {
      capturingRef = false;
      guide("参照物選択完了：実高さ(m)を入力してください");
      const Htrue = prompt("参照物の実高さ(m)を入力してください", "3.0");
      if (Htrue == null) { guide("参照物選択はキャンセルされました"); return; }
      const H = parseFloat(Htrue);
      if (!(H>0)) { toast("実高さは正の数で入力してください"); return; }
      // 推定高さ（参照物）を計算 → calScale
      const Hcalc = computeHeightFromClicks(refClicks);
      if (!isFinite(Hcalc) || Hcalc <= 0){
        toast("参照物の高さが計算できません。設定/クリックを確認してください。");
        return;
      }
      calScale = H / Hcalc;
      calBias = 0.0;
      toast(`校正完了: 実=${H.toFixed(3)} m / 推定=${Hcalc.toFixed(3)} m → cal_scale=${calScale.toFixed(6)}`);
    } else {
      guide("参照物：② 下端をタップ");
    }
  } else {
    if (clicks.length >= 2) return;
    clicks.push([xo, yo]);
    drawOverlay();
    if (clicks.length === 2){
      guide("選択完了：『高さを計算』を押してください");
    } else {
      const m = els.mode.value === 'cable' ? "② 同じ位置の地面をタップ（ケーブル直下）" : "② 電柱の下端をタップ";
      guide(m);
    }
  }
}

function startCaptureRef(){
  if (!els.photo.src) { toast("先に画像を選択してください"); return; }
  refClicks = []; capturingRef = true; drawOverlay();
  guide("参照物：① 上端 → ② 下端 を順にタップ");
}

function resetAll(){
  clicks = []; refClicks = []; capturingRef = false; calScale=1.0; calBias=0.0;
  drawOverlay(); updateGuide(); els.result.textContent = '';
}

/** --------- ガイド表示 --------- */
function guide(msg){ els.guide.textContent = msg; els.guide.className='status'; }
function toast(msg){ els.result.textContent = msg; els.result.className = 'status small'; }
function updateGuide(){
  if (!els.photo.src){ guide("① 画像を選択してください"); return; }
  if (els.mode.value === 'cable'){
    if (clicks.length===0) guide("① ケーブルの下端をタップ");
    else guide("② 同じ位置の地面をタップ（ケーブル直下）");
  } else {
    if (clicks.length===0) guide("① 電柱の上端をタップ");
    else guide("② 電柱の下端をタップ");
  }
}

/** --------- 座標変換（表示⇔原画像） --------- */
function scaleFactors(){
  // naturalWidth/Height はブラウザが解釈した表示向きのピクセル基準
  const nw = els.photo.naturalWidth || 1;
  const nh = els.photo.naturalHeight || 1;
  const rect = els.photo.getBoundingClientRect();
  const sx = nw / rect.width;
  const sy = nh / rect.height;
  // 返すのは: 表示→原画像 の拡大率（掛ける）
  return { sx, sy, rect };
}
function originalXY(ev){
  const { sx, sy, rect } = scaleFactors();
  const dx = ev.clientX - rect.left;
  const dy = ev.clientY - rect.top;
  const xo = dx * sx, yo = dy * sy;
  return [xo, yo];
}
function dispXY([xo, yo]){
  const { sx, sy, rect } = scaleFactors();
  return [ xo / sx, yo / sy ];
}

/** --------- 高さ計算（FOV法） --------- */
/* Python版の _height_fov と同等の考え方
   - センサー高さ(mm), 焦点距離(mm)
   - 画像中心からの縦オフセットを角度に変換（atan(y_mm/f))
   - 上下2点の角度差から高さ = D*(tan(θ1)-tan(θ2)) を求める
   - ピッチ角は両者から同じだけ差し引き（水平化）
*/
function computeHeightFromClicks(pts){
  if (pts.length < 2) return NaN;
  const distance = parseFloat(els.distance.value);
  const focal = parseFloat(els.focal.value);
  const sensorH = parseFloat(els.sensorH.value);
  const pitchDeg = parseFloat(els.pitch.value) || 0;

  if (!(distance>0 && focal>0 && sensorH>0)) return NaN;

  const w = els.photo.naturalWidth;
  const h = els.photo.naturalHeight;
  const sensorW = sensorH * (w/h);
  const mmPerPxX = sensorW / w;
  const mmPerPxY = sensorH / h;
  const cx = w/2, cy = h/2;

  function thetaFromPx(xp, yp){
    const x_mm = (xp - cx) * mmPerPxX;
    const y_mm = (yp - cy) * mmPerPxY;
    const yn = y_mm / focal; // 縦方向のみ
    return Math.atan(yn);
  }
  let thetaTop = thetaFromPx(...pts[0]);
  let thetaBottom = thetaFromPx(...pts[1]);
  const pitch = pitchDeg * Math.PI / 180;
  thetaTop -= pitch; thetaBottom -= pitch;

  const height = Math.abs( distance * (Math.tan(thetaTop) - Math.tan(thetaBottom)) );
  return height;
}

function onCalc(){
  if (!els.photo.src){ toast("画像がありません"); return; }
  if (clicks.length < 2){ toast("2点をタップしてください"); return; }
  const h = computeHeightFromClicks(clicks);
  if (!isFinite(h)){ toast("入力値（距離/焦点距離/センサー高さ）を確認してください"); return; }
  const hadj = calScale * h + calBias;

  const targetName = (els.mode.value === 'cable') ? "ケーブル高さ" : "電柱高さ";
  const line1 = `${targetName}: ${hadj.toFixed(2)} m`;
  const ok = hadj >= SAFE_THRESHOLD_M;
  els.guide.textContent = ok
    ? `判定：問題なし（${targetName} ${hadj.toFixed(2)} m ≥ ${SAFE_THRESHOLD_M.toFixed(2)} m）`
    : `判定：警告（${targetName} ${hadj.toFixed(2)} m ≤ ${SAFE_THRESHOLD_M.toFixed(2)} m）`;
  els.guide.className = 'status ' + (ok ? 'ok' : 'warn');

  els.result.textContent = [
    line1,
    `しきい値: ${SAFE_THRESHOLD_M.toFixed(2)} m`,
    calScale !== 1.0 ? `※ 校正適用中（cal_scale=${calScale.toFixed(6)}）` : ''
  ].filter(Boolean).join(' / ');
  els.result.className = 'status small';
}
